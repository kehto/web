  windowId: string,
  event: NostrEvent,
  topic: string,
  services: ServiceRegistry,
  sendToNapplet: SendToNapplet,
): boolean {
  const colonIndex = topic.indexOf(':');

**Section 1: ServiceHandler Interface Change (SVC-01)**

1.1 **Background** -- The ServiceHandler interface in `packages/runtime/src/types.ts` (line 486) is the contract between the runtime dispatch layer and all service implementations. Currently:
- `handleMessage(windowId: string, message: unknown[], send: (msg: unknown[]) => void): void`
- Services receive NIP-01 arrays (e.g., `['EVENT', event]`) and respond with NIP-01 arrays (e.g., `['OK', id, true, '']` or `['EVENT', subId, event]`)
- The runtime's `routeServiceMessage()` in `service-dispatch.ts` wraps IPC_PEER events as `['EVENT', event]` before calling `handleMessage`
- Some services (relay-pool, cache, coordinated-relay) receive raw NIP-01 verbs directly: `['REQ', subId, ...filters]`, `['CLOSE', subId]`, `['EVENT', event]`

Explain that this is Pitfall 5 from PITFALLS.md: after the runtime migrates to NUB envelope dispatch (RUNTIME-MIGRATION.md section 1), services will receive `NappletMessage` objects but still expect `unknown[]` arrays, causing silent failures.

1.2 **Old vs New Interface** -- Side-by-side:

Old (current):
```typescript
export interface ServiceHandler {
  descriptor: ServiceDescriptor;
  handleMessage(windowId: string, message: unknown[], send: (msg: unknown[]) => void): void;
  onWindowDestroyed?(windowId: string): void;
}
```

New (target):
```typescript
export interface ServiceHandler {
  descriptor: ServiceDescriptor;
  handleMessage(
    windowId: string,
    message: NappletMessage,
    send: (msg: NappletMessage) => void,
  ): void;
  onWindowDestroyed?(windowId: string): void;
}
```

Document the key changes:
- `message: unknown[]` becomes `message: NappletMessage` -- a typed JSON envelope `{ type: string, id?: string, ... }`
- `send: (msg: unknown[]) => void` becomes `send: (msg: NappletMessage) => void` -- responses are envelope objects
- NappletMessage comes from `@napplet/core` -- the type is `{ type: string } & Record<string, unknown>`
- The `unknown[]` type provided zero type safety; the new interface catches format mismatches at compile time

1.3 **service-dispatch.ts Rewrite** -- Document the routing change:

Old `routeServiceMessage()`:
- Extracts topic from NIP-01 event tags (`event.tags.find(t => t[0] === 't')[1]`)
- Routes by topic prefix before colon (`audio:register` -> prefix `audio` -> services['audio'])
- Wraps as `['EVENT', event]` before calling `handleMessage`
- Returns boolean indicating whether a service handled the message

New routing approach:
- Receives `NappletMessage` directly from the NUB dispatch layer
- Routes by `message.type` domain prefix (`message.type.split('.')[0]`)
- For IFC-routed services (audio, notifications): receives `ifc.emit` messages, extracts topic from `message.topic`, routes by topic prefix
- Passes the envelope object directly to `handleMessage` -- no wrapping
- The `routeServiceMessage` function signature changes: `(windowId, message: NappletMessage, services, sendToNapplet)` -- no more `event` or `topic` parameters

Show the new routing pseudocode:
```typescript
// For NUB-domain services (signer: handles signer.* messages directly)
const domain = message.type.split('.')[0];
const handler = services[domain];
if (handler) { handler.handleMessage(windowId, message, send); return true; }

// For IFC-routed services (audio, notifications: receive ifc.emit with topic prefix)
if (message.type === 'ifc.emit' && message.topic) {
  const prefix = message.topic.split(':')[0];
  const handler = services[prefix];
  if (handler) { handler.handleMessage(windowId, message, send); return true; }
}
```

1.4 **SendToNapplet Signature** -- Document the outbound side:
- Current: `SendToNapplet = (windowId: string, msg: unknown[]) => void`
- Target: `SendToNapplet = (windowId: string, msg: NappletMessage | unknown[]) => void` (widened for dual-mode, per RUNTIME-MIGRATION.md section 3.7)
- The `send` callback in service handlers narrows this to `(msg: NappletMessage) => void` since services only produce envelope responses under NIP-5D

1.5 **ServiceDescriptor** -- Note that ServiceDescriptor itself is UNCHANGED:
```typescript
interface ServiceDescriptor { name: string; version: string; description?: string; }
```
The descriptor is metadata only -- it does not participate in message routing under NIP-5D. Service discovery uses `services.has()` (synchronous, injected at creation) rather than kind 29010 queries (see RUNTIME-MIGRATION.md section 3.6).

1.6 **Dual-Mode Transition Strategy** -- Document how to handle the transition:
- Phase 1 (dual-mode): `handleMessage` accepts `NappletMessage | unknown[]` -- services check `Array.isArray(message)` to detect legacy format and use a compat wrapper
- Phase 2 (NIP-5D only): `handleMessage` accepts `NappletMessage` only -- array path removed
- The compat wrapper converts legacy arrays to NappletMessage for internal processing, so service logic is written only once against the envelope format
- This matches the runtime's own dual-mode strategy (RUNTIME-MIGRATION.md section 1.3)
