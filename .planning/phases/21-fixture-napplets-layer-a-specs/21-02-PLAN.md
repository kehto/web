// Import + use
import '@napplet/shim';   // installs window.napplet — required first
import {
  identity, ipc, notify, relay, storage, // namespaces (one per domain)
  type EventTemplate,                      // for relay templates
  type NostrEvent,                         // return type from relay.publish
  type Subscription,                       // return type from ipc.on
} from '@napplet/sdk';

// identity (read-only — always returns; runtime fallback emits .error if no signer)
identity.getPublicKey(): Promise<string>
identity.getProfile():   Promise<{ name?, displayName?, about?, picture?, banner?, nip05?, lud16?, website? } | null>

// ipc (inter-napplet pubsub via runtime ifcSubscriptions map)
ipc.emit(topic: string, extraTags?: string[][], content?: string): void
ipc.on(topic: string, callback: (payload: unknown, event: NostrEvent) => void): Subscription

// notify (runtime fallback handles notify.send + notify.permission.request)
notify.send({ title: string, body?: string, ... }): Promise<{ notificationId: string }>
notify.dismiss(notificationId: string): void

// relay (publishes via shell hooks — relay pool is mocked)
relay.publish(template: EventTemplate, options?: { relay?: boolean }): Promise<NostrEvent>
relay.publishEncrypted(template: EventTemplate, recipient: string, encryption: 'nip44' | 'nip04' = 'nip44'): Promise<NostrEvent>
relay.subscribe(filters, onEvent, onEose, options?): Subscription
relay.query(filters): Promise<NostrEvent[]>

// storage (localStorage-backed via shell hooks-adapter, scoped per napplet identity)
storage.getItem(key: string): Promise<string | null>
storage.setItem(key: string, value: string): Promise<void>
storage.removeItem(key: string): Promise<void>
storage.keys(): Promise<string[]>

// EventTemplate
interface EventTemplate {
  kind: number;
  content: string;
  tags: string[][];
  created_at: number;
}

// NOTE on theme: @napplet/sdk does NOT expose a theme namespace (verified — only
// type re-exports from @napplet/nub-theme). The runtime DOES handle theme.get
// envelopes natively (packages/runtime/src/runtime.ts:1048) and returns the
// canonical DEFAULT theme as a fallback when no 'theme' service is registered.
// For nub-theme fixture, we use shell-bridge's theme.changed push (see Plan 21-03)
// AND optionally fixture can call theme.get directly via __injectEnvelope__ from
// the spec — fixture itself does NOT need to call theme.get; spec drives it.
//
// However, for the fixture to RECEIVE theme.changed pushes from the shell, it
// must register a listener. Since SDK has no theme namespace, the fixture has TWO
// allowed paths:
//   (a) Use ipc.on('__theme__', ...) — NOT a real shell broadcast topic
//   (b) The harness spec injects a theme envelope directly to the fixture window
//       via __injectEnvelope__, and the fixture's last-received nub message is
//       captured by the harness via __getNubMessage__('test-napplet-N', 'theme.changed')
//
// Decision for this plan: nub-theme fixture is OUTBOUND-ONLY. It calls theme.get
// via the SDK fallback path (which doesn't exist — see workaround below) OR more
// simply, it calls storage.getItem on init to satisfy AUTH and exposes a #nub-status
// sentinel. The Layer-A nub-theme.spec.ts (Plan 21-03) drives EVERYTHING via
// __injectEnvelope__ + __getNubMessage__ on the fixture's windowId — fixture is
// just a loadable shell that the runtime can route theme envelopes to/from.
//
// nub-theme main.ts pattern: import shim + sdk.storage; await storage.getItem('nub-auth-probe')
// to gate AUTH; set #nub-status to 'authenticated'. That is sufficient — spec drives the rest.

Fixture template (mirror of `apps/demo/napplets/composer/`):
- `package.json` — `@napplet/shim` + `@napplet/sdk` link deps; vite + typescript devDeps; `dev`/`build`/`preview` scripts
- `vite.config.ts` — `nip5aManifest({ nappletType: '<fixture-name>' })`
- `tsconfig.json` — strict, ESNext, DOM lib
- `index.html` — minimal: meta `napplet-napp-type` matching nappletType, body with status div + per-domain DOM sentinels, `<script type="module" src="./src/main.ts">`
- `src/main.ts` — import shim + sdk namespace, do one or two SDK calls on init, set status sentinel

DOM sentinel convention (used by Plans 21-03 specs):
- ALL fixtures: `<div id="nub-status">connecting...</div>` — flips to `'authenticated'` after first SDK call resolves, then to a domain-specific success sentinel or `'denied:<reason>'`
- nub-identity: `<div id="nub-pubkey">` (truncated pubkey or 'no-pubkey'), `<div id="nub-profile-name">` (profile name or empty)
- nub-ifc: `<div id="nub-ifc-log">` (one line per received ipc.event), `<button id="nub-ifc-emit-btn">Emit</button>`
- nub-notify: `<div id="nub-notif-id">` (notification id from notify.send.result)
- nub-relay: `<div id="nub-event-id">` (event id from relay.publish), `<button id="nub-encrypt-btn">Encrypt</button>`, `<input id="nub-recipient">`
- nub-storage: `<div id="nub-storage-value">` (value read back via storage.getItem)
- nub-theme: `<div id="nub-theme-bg">` (last theme.colors.background received) — populated by spec-driven envelope only

Anti-feature reminder (CRITICAL):
- ZERO `window.addEventListener('message')` in any fixture src/main.ts — no SDK gap exemption. Use `ipc.on()` for cross-napplet IFC and SDK methods only.
- ZERO `window.nostr`, `signer-service`, `BusKind`, `kind 29001/29002`, `core-compat` consumers.

     * nub-identity fixture — exercises identity.getPublicKey + identity.getProfile (E2E-09).
     *
     * Layer-A spec drives this fixture via window.__loadNapplet__('nub-identity') from
     * tests/e2e/nub-identity.spec.ts. Spec asserts:
     *   - #nub-status flips to 'authenticated' then to 'pubkey:<truncated>' or 'denied:*'
     *   - #nub-pubkey contains a truncated hex pubkey OR 'no-pubkey' sentinel
     *   - The harness's __getNubMessage__(windowId, 'identity.getPublicKey') returns a
     *     non-null envelope after fixture init (proves the request was dispatched)
     *
     * Anti-features (hard-enforced): NO raw window.addEventListener, NO NIP-01 arrays,
     *   NO window.nostr, NO BusKind, NO signer-service.
     */
    import '@napplet/shim';
    import { identity } from '@napplet/sdk';

    const statusEl = document.getElementById('nub-status')!;
    const pubkeyEl = document.getElementById('nub-pubkey')!;
    const profileNameEl = document.getElementById('nub-profile-name')!;

    function fmt(err: unknown, fb: string): string {
      return err instanceof Error && err.message ? err.message : (typeof err === 'string' && err.length > 0 ? err : fb);
    }

    function truncate(pk: string): string {
      if (!pk || pk.length === 0) return 'no-pubkey';
      return `${pk.slice(0, 8)}...${pk.slice(-4)}`;
    }

      try {
        // First SDK call gates AUTH. Runtime fallback returns
        // { type: 'identity.getPublicKey.error', error: 'no signer configured' } when
        // no signer is wired in the harness — we treat that as a successful AUTH because
        // the await rejects with the error message, proving the envelope round-tripped.
        const pubkey = await identity.getPublicKey();
        statusEl.textContent = 'authenticated';
        pubkeyEl.textContent = truncate(pubkey);

        try {
          const profile = await identity.getProfile();
          profileNameEl.textContent = profile?.name ?? '';
        } catch {
          // identity.getProfile is allowed to fail without poisoning the spec — its purpose
          // is only to demonstrate a second envelope dispatch. Spec does not assert profile content.
        }

        statusEl.textContent = `pubkey:${truncate(pubkey)}`;
      } catch (err) {
        // No signer / denial — still proof of AUTH + envelope round-trip.
        statusEl.textContent = `denied:${fmt(err, 'identity:read')}`;
        pubkeyEl.textContent = 'no-pubkey';
      }
