**Canonical per-domain proxy shape:**
```typescript
// packages/shell/src/identity-proxy.ts (pattern — applies to all 5)

import type { Runtime } from '@kehto/runtime';
import type { NappletMessage } from '@napplet/core';
import type { OriginRegistry } from './origin-registry.js'; // (or an interface — origin registry today is a singleton)

export interface IdentityProxyDeps {
  /** The runtime engine that owns identity.* dispatch (Plan 12-03). */
  runtime: Runtime;
  /** Origin registry for resolving windowId -> iframe Window. */
  originRegistry: { getIframeWindow(windowId: string): Window | null };
}

export interface IdentityProxy {
  /** Route a napplet-originated identity.* envelope into the runtime. */
  dispatch(windowId: string, envelope: NappletMessage): void;
  /** Push a shell-initiated identity-domain envelope into a napplet iframe. */
  emit(windowId: string, envelope: NappletMessage): void;
}

export function createIdentityProxy(deps: IdentityProxyDeps): IdentityProxy {
  return {
    dispatch(windowId, envelope) {
      // Delegation: the runtime already owns identity.* dispatch after Plan 12-03.
      deps.runtime.handleMessage(windowId, envelope);
    },
    emit(windowId, envelope) {
      const win = deps.originRegistry.getIframeWindow(windowId);
      if (win) win.postMessage(envelope, '*');
    },
  };
}
```

**Keys-forwarder shape:**
```typescript
// packages/shell/src/keys-forwarder.ts

import type { NappletMessage } from '@napplet/core';
import type { SessionEntry } from './types.js';

export interface KeysForwarderDeps {
  originRegistry: { getIframeWindow(windowId: string): Window | null };
  sessionRegistry: { getAllEntries(): SessionEntry[] };
  /** Return true if the given napplet pubkey has the 'keys:forward' cap. */
  hasKeysForwardCap(pubkey: string): boolean;
  /** Optional: attach to a specific target (defaults to window). Useful for tests. */
  target?: EventTarget;
}

export interface KeysForwarder {
  /** Detach the keydown listener and release resources. */
  destroy(): void;
}

export function createKeysForwarder(deps: KeysForwarderDeps): KeysForwarder {
  const target = deps.target ?? window;
  const listener = (ev: Event): void => {
    const ke = ev as KeyboardEvent;
    const entries = deps.sessionRegistry.getAllEntries();
    for (const entry of entries) {
      if (!deps.hasKeysForwardCap(entry.pubkey)) continue;
      const iframe = deps.originRegistry.getIframeWindow(entry.windowId);
      if (!iframe) continue;
      const envelope: NappletMessage & {
        key: string; code: string;
        ctrl: boolean; alt: boolean; shift: boolean; meta: boolean;
      } = {
        type: 'keys.forward',
        key: ke.key, code: ke.code,
        ctrl: ke.ctrlKey, alt: ke.altKey, shift: ke.shiftKey, meta: ke.metaKey,
      };
      iframe.postMessage(envelope, '*');
    }
  };
  target.addEventListener('keydown', listener);
  return {
    destroy(): void {
      target.removeEventListener('keydown', listener);
    },
  };
}
```
