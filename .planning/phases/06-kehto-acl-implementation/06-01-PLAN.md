<objective>
Implement the NIP-5D identity key schema change and ACL state migration utility for @kehto/acl.

Purpose: The old 3-part composite key (`pubkey:dTag:hash`) uses an AUTH-derived pubkey that NIP-5D eliminates. This plan updates the key format to `dTag:hash` and provides a pure migration function for converting persisted ACL state from old to new format.

Output: Updated `toKey()`, optional `Identity.pubkey`, new `migrateAclState()` function, and comprehensive tests for all changed/new code.
</objective>

```typescript
export function toKey(identity: Identity): string {
  return `${identity.pubkey}:${identity.dTag}:${identity.hash}`;
}

export function check(state: AclState, identity: Identity, cap: number): boolean {
  const key = toKey(identity);
  const entry = state.entries[key];
  if (!entry) return state.defaultPolicy === 'permissive';
  if (entry.blocked) return false;
  return (entry.caps & cap) !== 0;
}
```
