    ```yaml
    # Phase 23 (v1.4): CI-02 requirement
    name: Unit Tests

    on:
      push:
      pull_request:

    concurrency:
      group: ${{ github.workflow }}-${{ github.ref }}
      cancel-in-progress: true

    jobs:
      unit:
        name: Vitest
        runs-on: ubuntu-latest
        steps:
          - name: Checkout
            uses: actions/checkout@v4

          - name: Setup pnpm
            uses: pnpm/action-setup@v3
            with:
              version: 10

          - name: Setup Node.js
            uses: actions/setup-node@v4
            with:
              node-version: '20'
              cache: 'pnpm'

          - name: Install dependencies
            run: pnpm install --frozen-lockfile

          - name: Run unit tests
            run: pnpm test
    ```

    Notes for the executor:

    - Do NOT add a build step before `pnpm test` — Vitest runs against TypeScript source via the project's vite config; package builds are not required for unit tests. (If turbo's `test` task graph declares a build dependency, turbo will handle it transparently.)
    - Do NOT add e2e steps — that is CI-03 (Plan 23-03).
    - Do NOT add coverage or upload-artifact steps — out of scope for this phase.
