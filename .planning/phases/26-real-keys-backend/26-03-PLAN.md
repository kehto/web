```ts
/**
 * Hotkey-chord demo napplet — exercises real keys backend (KEYS-03, Phase 26).
 *
 * Per CONTEXT.md Area 3 + Area 1:
 *   - On init: runs the D-04 AUTH probe (storage.getItem), then dispatches
 *     a keys.registerAction envelope subscribing to 'Ctrl+Shift+K' via the
 *     real @kehto/services keys-service (Plan 26-01 document-level listener).
 *   - On keys.forward: increments #hotkey-chord-count and updates #hotkey-chord-last.
 *   - #hotkey-chord-status transitions: 'connecting...' → 'authenticated' → 'subscribed'
 *   - After the Playwright spec (Plan 26-04) dispatches Ctrl+Shift+K on the host page,
 *     the service's document keydown listener fires its onForward callback, the shell
 *     fans out keys.forward envelopes, this napplet receives one and updates the DOM.
 *
 * Anti-features (enforced per v1.4 milestone):
 *   - zero window.addEventListener('message', ...) — uses @napplet/sdk exclusively
 *   - zero window.nostr / signer-service / BusKind / kind 29001|29002
 */
import '@napplet/shim';
import { storage } from '@napplet/sdk';

const statusEl = document.getElementById('hotkey-chord-status')!;
const countEl = document.getElementById('hotkey-chord-count')!;
const lastEl = document.getElementById('hotkey-chord-last')!;
const logEl = document.getElementById('hotkey-chord-log')!;

function setStatus(text: string, color: 'gray' | 'green' | 'red' = 'gray'): void {
  statusEl.textContent = text;
  statusEl.style.color = color === 'green' ? '#39ff14' : color === 'red' ? '#ff3b3b' : '#888';
}

function log(text: string): void {
  const div = document.createElement('div');
  div.className = 'hotkey-log-entry';
  const time = new Date().toLocaleTimeString('en', {
    hour12: false, hour: '2-digit', minute: '2-digit', second: '2-digit',
  });
  div.textContent = `${time} ${text}`;
  logEl.appendChild(div);
  logEl.scrollTop = logEl.scrollHeight;
}

function formatChord(ev: { key: string; ctrl: boolean; alt: boolean; shift: boolean; meta: boolean }): string {
  const parts: string[] = [];
  if (ev.ctrl) parts.push('Ctrl');
  if (ev.alt) parts.push('Alt');
  if (ev.shift) parts.push('Shift');
  if (ev.meta) parts.push('Meta');
  parts.push(ev.key.length === 1 ? ev.key.toUpperCase() : ev.key);
  return parts.join('+');
}

// ─── Envelope dispatch + listen ──────────────────────────────────────────────
// The @napplet/shim module installs a message listener on `window` and exposes
// `globalThis.nappletSend()` + a shim-managed dispatch surface. We use the
// shim's standard dispatch pattern (same surface @napplet/sdk's relay / storage
// helpers use). If @napplet/sdk exports a keys.registerAction helper, use it;
// otherwise the shim's global dispatch surface is the canonical fallback — see
// how relay.subscribe / storage.getItem funnel through the shim.
//
// Per KEYS-03: defaultKey MUST be 'Ctrl+Shift+K' (CONTEXT.md <specifics>).

declare const globalThis: typeof window & {
  nappletSend?: (msg: unknown) => void;
};

let deliveryCount = 0;
const REGISTER_ID = `hotkey-chord-register-${Math.random().toString(36).slice(2, 10)}`;

function onKeysForward(msg: { key: string; code: string; ctrl: boolean; alt: boolean; shift: boolean; meta: boolean }): void {
  deliveryCount += 1;
  countEl.textContent = String(deliveryCount);
  const chordStr = formatChord(msg);
  lastEl.textContent = chordStr;
  log(`keys.forward received — ${chordStr}`);
}

function onRegisterResult(msg: { type: string; actionId?: string; binding?: string }): void {
  setStatus('subscribed', 'green');
  log(`keys.registerAction.result — actionId=${msg.actionId ?? 'unknown'} binding=${msg.binding ?? 'none'}`);
}

// Shim-managed inbound envelope listener: the @napplet/shim installs one
// window.addEventListener('message', ...) on the napplet's BEHALF; we hook its
// message handoff via the standard shim event bus. We MUST NOT attach our own
// window.addEventListener('message', ...) — that violates the v1.4 anti-feature.
//
// @napplet/shim exposes its inbound bus as a plain EventTarget on a global
// `__nappletShim__` symbol (check napplet/packages/shim/src/index.ts to confirm
// the exact accessor). The executor verifies the actual shim API surface at
// implementation time and uses whichever accessor the shim ships.

// CANONICAL APPROACH — if @napplet/sdk exposes a `keys` helper, use it:
//   import { keys } from '@napplet/sdk';
//   keys.onForward(onKeysForward);
//   keys.registerAction({ id: 'hotkey-chord.demo', label: 'Demo Chord', defaultKey: 'Ctrl+Shift+K' })
//     .then(onRegisterResult);
//
// FALLBACK APPROACH — if no keys helper ships in the SDK, use the shim's
// envelope handoff (no raw window.addEventListener):
import { onNappletMessage, sendNappletMessage } from '@napplet/sdk';

onNappletMessage((envelope: { type: string } & Record<string, unknown>) => {
  if (envelope.type === 'keys.forward') {
    onKeysForward(envelope as unknown as { key: string; code: string; ctrl: boolean; alt: boolean; shift: boolean; meta: boolean });
    return;
  }
  if (envelope.type === 'keys.registerAction.result' && (envelope as { id?: string }).id === REGISTER_ID) {
    onRegisterResult(envelope as { type: string; actionId?: string; binding?: string });
    return;
  }
});

async function init(): Promise<void> {
  // D-04 init pattern: storage.getItem probe gates on AUTH completion without
  // needing a state:read grant (identical to feed napplet).
  try {
    await storage.getItem('hotkey-chord-auth-probe');
  } catch { /* state:read denial is expected — AUTH still completed */ }
  setStatus('authenticated', 'green');
  log('AUTH complete — dispatching keys.registerAction (Ctrl+Shift+K)');

  sendNappletMessage({
    type: 'keys.registerAction',
    id: REGISTER_ID,
    action: {
      id: 'hotkey-chord.demo',
      label: 'Hotkey-Chord Demo',
      defaultKey: 'Ctrl+Shift+K',
    },
  });
}

init().catch((err) => {
  if (statusEl.textContent === 'connecting...') {
    setStatus('auth failed', 'red');
    log(`init failed — ${err instanceof Error ? err.message : String(err)}`);
  }
});
```
