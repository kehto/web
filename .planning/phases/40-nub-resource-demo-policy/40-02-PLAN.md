      ```ts
      const GRANTED_URL = 'http://localhost:5174/demo-data.json';
      const DENIED_URL  = 'https://untrusted.example/'; // D4: RFC-2606 reserved

      function newRequestId(): string { return `res-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`; }

      async function init(): Promise<void> {
        setStatus('requesting...', 'gray');
        const grantedId = newRequestId();
        const deniedId  = newRequestId();

        // Install envelope listener BEFORE dispatching so we catch both outcomes.
        // If @napplet/sdk exposes an onEnvelope(...)/subscribe helper, use it.
        // Otherwise wire via @napplet/shim's onMessage (mirror config-demo).
        onEnvelope((env: ResourceBytesResult | ResourceBytesError) => {
          if (env.type === 'resource.bytes.result' && env.requestId === grantedId) {
            // Decode base64 → UTF-8 → JSON parse for display.
            try {
              const binary = atob(env.bodyBase64);
              const bytes = Uint8Array.from(binary, c => c.charCodeAt(0));
              const text = new TextDecoder('utf-8').decode(bytes);
              const parsed = JSON.parse(text);
              grantedEl.textContent = JSON.stringify(parsed, null, 2);
              log(`granted status=${env.status} bytes=${bytes.length}`);
            } catch (err) {
              grantedEl.textContent = `decode error: ${(err as Error).message}`;
            }
          }
          if (env.type === 'resource.bytes.error' && env.requestId === grantedId) {
            grantedEl.textContent = `unexpected error on granted fetch: code=${env.code} message=${env.message}`;
            log(`granted error ${env.code}`);
          }
          if (env.type === 'resource.bytes.error' && env.requestId === deniedId) {
            deniedEl.textContent = `code=${env.code} message=${env.message}`;
            log(`denied ${env.code}`);
          }
          if (env.type === 'resource.bytes.result' && env.requestId === deniedId) {
            // Should never happen — denied origin must not reach fetch.
            deniedEl.textContent = `PROTOCOL VIOLATION: denied origin returned ${env.status}`;
          }
        });

        dispatch({ type: 'resource.bytes', requestId: grantedId, url: GRANTED_URL });
        dispatch({ type: 'resource.bytes', requestId: deniedId,  url: DENIED_URL });
        setStatus('dispatched 2 requests', 'gray');
      }
      ```

      ```ts
      // Host `fetch` = native browser fetch + AbortController + 10s timeout (D5).
      // Origin gating via connectStore.getOrigins (dTag-scoped grants from Phase 39).
      // H-03 guard (Wave 1): all 4 options are required; factory throws if any is missing.
      async function hostFetch(
        url: string,
        init: { method?: string; headers?: Record<string, string>; signal: AbortSignal },
      ): Promise<Response> {
        const timeoutController = new AbortController();
        const timeoutId = setTimeout(() => timeoutController.abort(), 10_000);
        // Compose: forward both abort signals into fetch.
        const composite = new AbortController();
        for (const src of [init.signal, timeoutController.signal]) {
          src.addEventListener('abort', () => composite.abort(), { once: true });
        }
        try {
          return await fetch(url, { method: init.method, headers: init.headers, signal: composite.signal });

          clearTimeout(timeoutId);
      ```
