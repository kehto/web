    ```ts
    /** [migration JSDoc header as specified in <interfaces>] */

    import type { NappletMessage } from '@napplet/core';
    import type { ServiceHandler, Signer } from '@kehto/runtime';
    import type {
      IdentityGetPublicKeyResultMessage,
      IdentityGetRelaysResultMessage,
      IdentityGetProfileResultMessage,
      IdentityGetFollowsResultMessage,
      IdentityGetListResultMessage,
      IdentityGetZapsResultMessage,
      IdentityGetMutesResultMessage,
      IdentityGetBlockedResultMessage,
      IdentityGetBadgesResultMessage,
    } from '@napplet/nub-identity';

    export interface IdentityServiceOptions {
      /** Return the NIP-07-compatible signer (or null) used to resolve
       *  identity.getPublicKey / identity.getRelays. */
      getSigner: () => Signer | null;
    }

    export function createIdentityService(options: IdentityServiceOptions): ServiceHandler {
      return {
        descriptor: { name: 'identity', version: '1.0.0', description: 'NIP-5D identity NUB reference handler' },
        handleMessage(windowId, message, send) {
          const id = (message as NappletMessage & { id?: string }).id ?? '';
          function sendResult<T extends NappletMessage>(payload: Omit<T, 'id'> & { id?: string }): void {
            send({ id, ...payload } as NappletMessage);
          }
          function sendError(typeBase: string, error: string): void {
            send({ type: `${typeBase}.error`, id, error } as NappletMessage);
          }

          const signer = options.getSigner();

          switch (message.type) {
            case 'identity.getPublicKey': {
              if (!signer) { sendError('identity.getPublicKey','no signer configured'); return; }
              Promise.resolve(signer.getPublicKey?.())
                .then((pubkey) => sendResult<IdentityGetPublicKeyResultMessage>({
                  type: 'identity.getPublicKey.result', pubkey: pubkey as string,
                }))
                .catch((err: unknown) => sendError('identity.getPublicKey', (err as Error).message ?? 'failed'));
              return;
            }
            case 'identity.getRelays': {
              if (!signer) { sendError('identity.getRelays','no signer configured'); return; }
              Promise.resolve(signer.getRelays?.() ?? {})
                .then((relays) => sendResult<IdentityGetRelaysResultMessage>({
                  type: 'identity.getRelays.result', relays: relays as Record<string, { read: boolean; write: boolean }>,
                }))
                .catch((err: unknown) => sendError('identity.getRelays', (err as Error).message ?? 'failed'));
              return;
            }
            case 'identity.getProfile':
              sendResult<IdentityGetProfileResultMessage>({ type: 'identity.getProfile.result', profile: null });
              return;
            case 'identity.getFollows':
              sendResult<IdentityGetFollowsResultMessage>({ type: 'identity.getFollows.result', pubkeys: [] });
              return;
            case 'identity.getList':
              sendResult<IdentityGetListResultMessage>({ type: 'identity.getList.result', entries: [] });
              return;
            case 'identity.getZaps':
              sendResult<IdentityGetZapsResultMessage>({ type: 'identity.getZaps.result', zaps: [] });
              return;
            case 'identity.getMutes':
              sendResult<IdentityGetMutesResultMessage>({ type: 'identity.getMutes.result', pubkeys: [] });
              return;
            case 'identity.getBlocked':
              sendResult<IdentityGetBlockedResultMessage>({ type: 'identity.getBlocked.result', pubkeys: [] });
              return;
            case 'identity.getBadges':
              sendResult<IdentityGetBadgesResultMessage>({ type: 'identity.getBadges.result', badges: [] });
              return;
            default:
              sendError(message.type, `Unknown identity method: ${message.type}`);
          }
        },
        onWindowDestroyed() { /* no-op */ },
      };
    }
    ```

    ```ts
    function handleIdentityMessage(windowId: string, msg: NappletMessage): void {
      const identityService = serviceRegistry['identity'];
      if (identityService) {
        identityService.handleMessage(windowId, msg, (resp: NappletMessage) => {
          hooks.sendToNapplet(windowId, resp);
        });
        return;
      }
      // No identity service registered — fall back to hooks.auth.getSigner() for the
      // two read-only identity-mapping cases (getPublicKey/getRelays) so existing hosts
      // do not break. Other 7 actions return an empty/default payload so napplets get
      // a result envelope rather than a silent drop.
      const id = (msg as NappletMessage & { id?: string }).id ?? '';
      const action = msg.type.slice('identity.'.length);
      const signer = hooks.auth.getSigner();
      function sendResult(payload: Record<string, unknown>): void {
        hooks.sendToNapplet(windowId, { type: `${msg.type}.result`, id, ...payload } as NappletMessage);
      }
      switch (action) {
        case 'getPublicKey':
          if (!signer) { hooks.sendToNapplet(windowId, { type: 'identity.getPublicKey.error', id, error: 'no signer configured' } as NappletMessage); return; }
          Promise.resolve(signer.getPublicKey?.()).then((pubkey) => sendResult({ pubkey })).catch(() => hooks.sendToNapplet(windowId, { type: 'identity.getPublicKey.error', id, error: 'failed' } as NappletMessage));
          return;
        case 'getRelays':
          if (!signer) { hooks.sendToNapplet(windowId, { type: 'identity.getRelays.error', id, error: 'no signer configured' } as NappletMessage); return; }
          Promise.resolve(signer.getRelays?.() ?? {}).then((relays) => sendResult({ relays })).catch(() => hooks.sendToNapplet(windowId, { type: 'identity.getRelays.error', id, error: 'failed' } as NappletMessage));
          return;
        case 'getProfile':  sendResult({ profile: null }); return;
        case 'getFollows':  sendResult({ pubkeys: [] }); return;
        case 'getList':     sendResult({ entries: [] }); return;
        case 'getZaps':     sendResult({ zaps: [] }); return;
        case 'getMutes':    sendResult({ pubkeys: [] }); return;
        case 'getBlocked':  sendResult({ pubkeys: [] }); return;
        case 'getBadges':   sendResult({ badges: [] }); return;
        default:
          hooks.sendToNapplet(windowId, { type: `${msg.type}.error`, id, error: `Unknown identity action: ${action}` } as NappletMessage);
      }
    }
    ```
