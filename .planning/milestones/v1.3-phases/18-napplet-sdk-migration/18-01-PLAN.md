From @napplet/sdk:

```typescript
// Side-effect import installs window.napplet AND wires AUTH internally.
import '@napplet/shim';

// Named exports:
import { ipc, storage } from '@napplet/sdk';

// IPC namespace (for bot↔chat envelope traffic):
ipc.emit(topic: string, extraTags?: string[][], content?: string): void
ipc.on(topic: string, callback: (payload: unknown, event: NostrEvent) => void): Subscription

// Storage namespace (per-napplet scoped, async):
storage.getItem(key: string): Promise<string | null>
storage.setItem(key: string, value: string): Promise<void>
storage.removeItem(key: string): Promise<void>
storage.keys(): Promise<string[]>
```

**Key shape note** (verified in shim/src/index.ts line 105-108):
- `ipc.emit` content is a JSON-encoded string (the chat napplet emits `JSON.stringify({ text, timestamp })`).
- `ipc.on` callback receives `payload` already JSON-parsed by the shim.
- The shim auto-handles AUTH; do not write a manual handshake.

import { ipc, storage } from '@napplet/sdk';

async function init(): Promise<void> {
  // First SDK call that requires AUTH — if this resolves, the shim completed AUTH.

  // Now we know AUTH succeeded.

  statusEl.style.color = '#39ff14';
  // ... wire ipc.on, parse rules, etc.
}

init().catch((err) => {
  statusEl.textContent = 'auth failed';
  statusEl.style.color = '#ff3b3b';

});
