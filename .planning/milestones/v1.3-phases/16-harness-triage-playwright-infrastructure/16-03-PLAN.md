 * Inject a NIP-5D envelope into the runtime as if posted by the given napplet.
 * Synchronous — the envelope passes through relay.handleMessage immediately.
 * @param windowId - The harness-assigned window id for the napplet iframe.
 * @param envelope - A NIP-5D NappletMessage (must include `type: string`).
 */
__injectEnvelope__: (windowId: string, envelope: NappletMessage) => void;

/**
 * Snapshot the most recent NIP-5D envelope recorded for a napplet, optionally
 * filtered by envelope `type`. Returns a deep clone (structured-clone-safe).
 * @param windowId - The harness-assigned window id.
 * @param type - Optional envelope type filter (e.g., "relay.publish").
 * @returns The last matching NappletMessage, or null if none recorded.
 */
__getNubMessage__: (windowId: string, type?: string) => NappletMessage | null;

/**
 * List the currently registered service names from the runtime's service registry.
 * Includes both the initial set from createMockHooks and any services registered
 * via __registerService__ at runtime.
 * @returns An array of service names (e.g., ["identity", "notifications"]).
 */
__getServiceNames__: () => string[];

/**
 * Evaluate `handlerScript` in the harness context (no network/IO side effects
 * expected) and register the resulting handler with the runtime under `name`.
 * The script MUST evaluate to a ServiceHandler-shaped object.
 *
 * @example __registerService__('keys', '({ name: "keys", version: "0.1.0", handleMessage: () => null })')
 *
 * @param name - The service name to register.
 * @param handlerScript - A JS expression (not a full function body) that evaluates to a ServiceHandler.
 * @returns true if registration succeeded; false if the script evaluation threw.
 */
__registerService__: (name: string, handlerScript: string) => boolean;

/**
 * Unregister a service by name.
 * @param name - The service name previously registered.
 * @returns true if a service was removed; false if no such service existed.
 */
__unregisterService__: (name: string) => boolean;

/**
 * Snapshot the notification-service state, deep-cloned and filtered by windowId.
 * @param windowId - Optional filter; if omitted, returns all notifications.
 * @returns An array of serializable notification entries.
 */
__getNotifications__: (windowId?: string) => Array<{ id: string; title: string; body?: string; read: boolean }>;

/**
 * Set the pubkey returned by subsequent `identity.getPublicKey` requests.
 * Delegates to mockResult.setUserPubkey() under the hood.
 * @param pubkey - 64-char hex pubkey string.
 */
__setIdentityPubkey__: (pubkey: string) => void;

/**
 * Readiness flag per napplet — set to true by the harness AFTER the runtime

 * for this windowId's pubkey). Consumed by Plan 16-04's waitForNappletReady helper.
 * @param windowId - The napplet's window id.
 * @returns true if the napplet's session is acknowledged; false otherwise.

Wiring notes — where each implementation hooks into harness.ts:

1. `__injectEnvelope__`: reuse existing `__injectMessage__` path but wrap the envelope in a
   NIP-5D envelope kind (29005 — or whatever the canonical NIP-5D event kind is). Actually:
   NIP-5D envelopes use `postMessage(envelope)` directly (not NIP-01 array), per
   ARCHITECTURE.md + shell-bridge.ts. The correct injection path is:
     const iframe = nappletFrames.get(windowId);
     const event = new MessageEvent('message', { data: envelope, source: iframe.contentWindow, origin: 'null' });
     window.dispatchEvent(event);
   Note: the current `__injectMessage__` expects `data: unknown[]` (NIP-01 array). NIP-5D
   envelopes are plain objects with `type`. Do NOT collapse the two — add __injectEnvelope__
   as a sibling function with its own dispatch path.

2. `__getNubMessage__`: subscribe to the existing `tap` (createMessageTap) for recorded
   messages. Extend `tap` OR maintain a harness-local envelope log in this harness.ts.
   Since `tap` today only understands NIP-01 arrays (TappedMessage shape), add a NEW
   shadow log: `const envelopeLog = new Map<string, NappletMessage[]>()`. Populate it in
   the proxied postMessage interception that already runs in harness.ts — when
   `Array.isArray(msg)` is false AND msg has a `type` property, push a clone into
   envelopeLog.get(windowId). `__getNubMessage__` returns JSON.parse(JSON.stringify(last))
   for structured-clone safety (no method refs).

3. `__getServiceNames__`: maintain `const serviceShadow = new Set<string>()`. Populate
   initially from `Object.keys(mockResult.hooks.services ?? {})`. Update in
   __registerService__/__unregisterService__. Return `[...serviceShadow]`.

4. `__registerService__`: call `new Function('return (' + handlerScript + ')')()` to eval
   the script (NOT `eval()` directly — constructor keeps scope isolated). Validate the
   result has shape `{ name: string; version: string; handleMessage: function }` — if not,
   return false. On success call `relay.runtime.registerService(name, handler)` and update
   serviceShadow.

5. `__unregisterService__`: check if serviceShadow has `name`; if yes, call
   `relay.runtime.unregisterService(name)` and remove from shadow, return true; else false.

6. `__getNotifications__`: If mockResult's adapter exposes a notification service via
   `hooks.services?.notifications`, the service handler accumulates state internally.
   APPROACH: Intercept the `onChange` callback by passing a harness-controlled wrapper in
   createMockHooks (or hook into mockResult.callLog if notifications are logged there). If
   neither path is available, maintain an empty shadow array and document the
   limitation — Phase 19 (toaster napplet + notify-lifecycle spec) can extend this if needed.

   SAFE FALLBACK: Return [] if no notification adapter is present. This still closes E2E-04
   (the global exists and returns structured-clone-safe data); richer behavior lands in
   Phase 19 alongside the `toaster` napplet.

7. `__setIdentityPubkey__`: single-line delegation: `mockResult.setUserPubkey(pubkey)`.
   mockResult.setUserPubkey already exists per tests/helpers/mock-hooks.ts line 134.

8. `__nappletReady__`: Read `relay.runtime.sessionRegistry.getPubkey(windowId)` and return
   true if truthy. (The existing `__getNappPubkey__` already does this — __nappletReady__
   is explicitly a boolean-returning wrapper for cleaner use in waitForFunction.)

Structured-clone-safety discipline — EVERY new global's return value must satisfy:
  - Primitives (string, number, boolean, null, undefined) — always safe
  - Plain objects with primitive fields or nested plain objects/arrays — safe
  - NEVER return `ShellBridge`, `Runtime`, `MockHooksResult`, `Map`, `Set`, or any object
    with method references. All existing methods that return live refs (__getRelay__,
    __getMockHooks__) are kept but are known-unsafe for page.evaluate() return values.

     * Deep-cloned — safe to return from page.evaluate().
     */
    window.__getNubMessage__ = (windowId: string, type?: string): NappletMessage | null => {
      const arr = envelopeLog.get(windowId) ?? [];
      const filtered = type ? arr.filter(m => m.type === type) : arr;
      if (filtered.length === 0) return null;
      const last = filtered[filtered.length - 1];
      return JSON.parse(JSON.stringify(last)) as NappletMessage;
    };

    /**
     * Snapshot of currently registered service names.
     */
    window.__getServiceNames__ = (): string[] => [...serviceShadow];

    /**
     * Register a service with the runtime by evaluating `handlerScript` in the harness context.
     * `handlerScript` MUST be a JS expression that evaluates to a ServiceHandler
     * (e.g., `({ name: "x", version: "1.0", handleMessage: (w, m, send) => null })`).
     */
    window.__registerService__ = (name: string, handlerScript: string): boolean => {
      try {
        // eslint-disable-next-line no-new-func -- test-only eval path
        const handler = new Function(`return (${handlerScript});`)();
        if (!handler || typeof handler.handleMessage !== 'function') return false;
        relay.runtime.registerService(name, handler);
        serviceShadow.add(name);
        return true;
      } catch {
        return false;
      }
    };

    /**
     * Unregister a service by name.
     */
    window.__unregisterService__ = (name: string): boolean => {
      if (!serviceShadow.has(name)) return false;
      relay.runtime.unregisterService(name);
      serviceShadow.delete(name);
      return true;
    };

    /**
     * Snapshot the notification-service state, filtered by windowId.
     * Returns [] if no notification service is wired (Phase 19 extends this).
     */
    window.__getNotifications__ = (windowId?: string) => {
      const filtered = windowId
        ? notificationsShadow.filter(n => n.windowId === windowId)
        : notificationsShadow;
      return filtered.map(n => ({
        id: n.id,
        title: n.title,
        body: n.body,
        read: n.read,
      }));
    };

    /**
     * Override the mock signer's user pubkey.
     * Delegates to MockHooksResult.setUserPubkey().
     */
    window.__setIdentityPubkey__ = (pubkey: string): void => {
      mockResult.setUserPubkey(pubkey);
    };

    /**
     * Readiness flag per napplet: true if the runtime has acknowledged this napplet's
     * session (i.e., sessionRegistry has an entry for its pubkey).
     * Consumed by tests/e2e/helpers/wait-for-napplet-ready.ts (Plan 16-04).
     */
    window.__nappletReady__ = (windowId: string): boolean => {
      return Boolean(relay.runtime.sessionRegistry.getPubkey(windowId));
    };
