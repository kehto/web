// Top of harness.ts (with other imports):
import { createKeysService, createMediaService } from '@kehto/services';

// Inside window.__registerService__:
window.__registerService__ = (name: string, handlerScript: string): boolean => {
  try {
    // NEW: real-factory branch — literal string 'real' swaps in the reference
    // @kehto/services factory for this name.
    if (handlerScript === 'real') {
      let handler;
      if (name === 'keys') handler = createKeysService();
      else if (name === 'media') handler = createMediaService();
      else return false; // 'real' only supported for keys + media in Phase 28
      relay.runtime.registerService(name, handler);
      serviceShadow.add(name);
      return true;
    }
    // Existing eval path — unchanged.
    const handler = new Function(`return (${handlerScript});`)();
    if (!handler || typeof handler.handleMessage !== 'function') return false;
    relay.runtime.registerService(name, handler);
    serviceShadow.add(name);
    return true;
  } catch {
    return false;
  }
};
